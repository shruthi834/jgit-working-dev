public MapSpec(@JsonProperty("mapName") String mapName,
                   @JsonProperty("fhlbId") String fhlbId,
                   @JsonProperty("rowDefinitions") RowDefinition[] rowDefinitions,
                   @JsonProperty("codeLists") CodeList[] codeLists,
                   @JsonProperty("srcCode") String srcCode,
                   @JsonProperty(value = "gisMap", required = false, defaultValue = "N") String gisMap) {
        this.mapName = mapName;
        this.rowDefinitions = rowDefinitions;
        this.codeLists = codeLists;
        this.fhlbId = fhlbId;
        this.srcCode = srcCode;
        this.gisMap = gisMap;
    }
