import com.opencsv.CSVReader;
import com.opencsv.CSVWriter;

import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.util.List;

public class UpdateCSVBasedOnID {
    public static void main(String[] args) {
        String filePath = "your_file.csv"; // Replace with your CSV file path
        String targetID = "123"; // Replace with the ID you want to search for

        try {
            // Read the CSV file
            CSVReader csvReader = new CSVReader(new FileReader(filePath));
            List<String[]> data = csvReader.readAll();
            csvReader.close();

            // Find the record with the matching ID
            int idColumnIndex = 0; // Assuming the ID is in the first column

            for (String[] row : data) {
                if (row.length > idColumnIndex && row[idColumnIndex].equals(targetID)) {
                    // Update the data
                    int columnToUpdate = 1; // Example column to update
                    String newValue = "Updated Value";

                    row[columnToUpdate] = newValue;

                    break; // Exit the loop once the record is updated
                }
            }

            // Write the updated data back to the CSV file
            CSVWriter csvWriter = new CSVWriter(new FileWriter(filePath));
            csvWriter.writeAll(data);
            csvWriter.close();

            System.out.println("CSV file updated successfully.");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
