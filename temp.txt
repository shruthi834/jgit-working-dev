    .map(a -> {
        Optional<MyType> matchingB = listB.stream()
            .filter(b -> a.getType().equals(b.getType()))
            .findFirst();

        if (matchingB.isPresent()) {
            // Set a.getValue() to a particular value when getType is equal

        }

        return a;
    })
    .collect(Collectors.toList());
