public static void main(String[] args) {
    String csvFile = "C:\\Users\\ryeso\\Downloads\\listOfCustomers (2).csv";
    String targetId = "6"; // The ID you want to look up
    try (CSVReader csvReader = new CSVReaderBuilder(new FileReader(csvFile)).build()) {
        String[] header = csvReader.readNext(); // Read the header row
        if (header != null) {
            int idIndex = -1;
            for (int i = 0; i < header.length; i++) {
                if ("ID".equalsIgnoreCase(header[i])) {
                    idIndex = i;
                    break;
                }
            }
            if (idIndex >= 0) {
                String[] row;
                while ((row = csvReader.readNext()) != null) {
                    if (targetId.equals(row[idIndex])) {
                        // Print the complete row for the target ID
                        for (int i = 0; i < header.length; i++) {
                            System.out.println(header[i] + ": " + row[i]);
                        }
                        break; // Exit the loop after finding the target ID
                    }
                }
            }
        }
    } catch (IOException | CsvValidationException e) {
        e.printStackTrace();
    }
}
