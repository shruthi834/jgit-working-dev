  List<ArchivedFile> commitDetailsList = new ArrayList<>();
        try (Repository repository = Git.open(new File(stashRepository)).getRepository()) {
            Git git = new Git(repository);
            String publishedMapPath = stashPublished + className +".java";
            File fileSize = new File(stashRepository+"/"+publishedMapPath);
            Iterable<RevCommit> logs = git.log().addPath(publishedMapPath).call();
            List<RevCommit> logsAddedToList = new ArrayList<>();
            for(RevCommit commitLength :logs){
                logsAddedToList.add(commitLength);
            }
            for (RevCommit commit : logsAddedToList) {
                ArchivedFile archivedFile=new ArchivedFile(null,commit.getId().getName(),commit.getFullMessage(),FileUtils.sizeOf(fileSize),commit.getAuthorIdent().getWhen(),"N");
                commitDetailsList.add(archivedFile);
            }
        }
        catch (Exception e) {
            log.error("Error in retrieving the commit history : " , e);
        }
        return commitDetailsList;
